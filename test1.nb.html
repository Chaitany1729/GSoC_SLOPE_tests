Loading the SLOPE package and abalone dataset.

library(SLOPE)

X <- abalone$x
y <- abalone$y
n = dim(X)[1]
p = dim(X)[2]

Fiting SLOPE with BH method.

fit_SLOPE <- SLOPE(X, y, family = "poisson", lambda = "bh")

plot(fit_SLOPE)

plot(fit_SLOPE, x_variable = "step")

plot(fit_SLOPE, x_variable = "deviance_ratio")
